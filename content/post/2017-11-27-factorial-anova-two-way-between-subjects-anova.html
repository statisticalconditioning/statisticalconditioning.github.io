library(knitr)
library(AMCP)
library(car)

opts_chunk$set(echo = TRUE, comment = NULL, highlight = FALSE)
data(C7T5)
df <- within(C7T5, {
  Feedback <- factor(Feedback)
  Drug     <- factor(Drug)
})
Anova(lm(Score ~ Feedback*Drug, df))
